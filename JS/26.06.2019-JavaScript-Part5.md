## List the properties of an object:
```
var object = {
name :"Nhat",
yearofbirth :"1997",
City: "Đà Nẵng",
company :"Wakumo"
}
console.log(object.name)
console.log(object.yearofbirth)
console.log(object.City)
console.log(object.company)

```

## Delete the the subClass properties

``` 
var object = {
name :"Nhat",
yearofbirth :"1997",
City: "Da Nang",
company :"wakumo"
}
delete object.yearofbirth;
delete object.City;

console.log(object.name)
console.log(object.yearofbirth)
console.log(object.City)
console.log(object.company)


```
